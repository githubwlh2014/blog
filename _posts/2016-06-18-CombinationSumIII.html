{% highlight ruby %} public class Solution { private List> list=new ArrayList<>(); private List subList=new ArrayList<>(); private void combinationSum3(int start,int k,int n,int sum) { if(sum==n&&k==0) { list.add(new ArrayList<>(subList)); return ; } for(int i=start;sum+i<=n&&k>0&&i<=10-k;i++) { subList.add(i); combinationSum3(i+1,k-1,n,sum+i); subList.remove((int)(subList.size()-1)); } } public List> combinationSum3(int k, int n) { list.clear(); combinationSum3(1,k,n,0); return list; } } {% endhighlight %}
