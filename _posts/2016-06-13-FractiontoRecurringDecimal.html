模仿除法的运算过程，切记切记一定要把除数和被除数转化为long类型

{% highlight ruby %} public class Solution { public String fractionToDecimal(int numerato, int denominato) { if(numerato==0) return "0"; boolean signTag=true; if(numerato>=0&&denominato>0) signTag=true; else if(numerato<0&&denominato<0) signTag=true; else signTag=false; long numerator=Math.abs((long)numerato); long denominator=Math.abs((long)denominato); StringBuilder sb=new StringBuilder(Long.toString(numerator/denominator)); numerator=numerator%denominator; if(numerator!=0) sb.append('.'); Map map=new HashMap<>(); int i=sb.length(); while(numerator!=0&&!map.containsKey(numerator)) { map.put(numerator, i++); numerator*=10; sb.append((char)(numerator/denominator+'0')); numerator%=denominator; } if(numerator!=0) { sb.append(')'); sb.insert(map.get(numerator), "("); } if(!signTag) return "-"+sb.toString(); return sb.toString(); } } {% endhighlight %}
