查找二叉搜索树的第K小的元素，中序递归遍历

{% highlight ruby %} public class Solution { int count=0; boolean tag=false; int element=0; public void kthSmallestEle(TreeNode root,int k) { if(tag) return ; if(root.left!=null) kthSmallestEle(root.left,k); count++; if(count==k) { element=root.val; tag=true; return ; } if(root.right!=null) kthSmallestEle(root.right,k); } public int kthSmallest(TreeNode root, int k) { count=0; tag=false; kthSmallestEle(root,k); return element; } } {% endhighlight %}
