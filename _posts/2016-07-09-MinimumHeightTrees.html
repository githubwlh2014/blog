类似于拓扑排序，逐层去掉叶子结点，向根结点进行逼近

{% highlight ruby %} public class Solution { public List findMinHeightTrees(int n, int[][] edges) { if(n<1) return new ArrayList(); if(n==1) return new ArrayList(Arrays.asList(0)); List> list=new ArrayList<>(); for(int i=0;i()); for(int[] val:edges) { list.get(val[0]).add(val[1]); list.get(val[1]).add(val[0]); } Queue leaves=new LinkedList<>(); for(int i=0;i2) { n-=leaves.size(); Queue newLeaves=new LinkedList(); while(leaves.size()>0) { int i=leaves.poll(); int j=list.get(i).iterator().next(); list.get(j).remove(i); if(list.get(j).size()==1) newLeaves.offer(j); } leaves=newLeaves; } return (List)leaves; } } {% endhighlight %}
