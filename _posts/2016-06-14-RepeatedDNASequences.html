使用HashMap进行统计

{% highlight ruby %} public class Solution { public List findRepeatedDnaSequences(String s) { Map map=new HashMap<>(); List list=new ArrayList<>(); if(s==null) return list; for(int i=0;i<=s.length()-10;i++) { String str=s.substring(i,i+10); if(map.get(str)==(Integer)1) { list.add(str); } else if(map.get(str)==null) map.put(str, 0); map.put(str, map.get(str)+1); } return list; } } {% endhighlight %}
