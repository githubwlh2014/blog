两个索引slow、fast分别同时从数组为0开始，先走fast经过元素相加，当和不小于s时走slow，在这个过程中统计满足条件的最小步长

{% highlight ruby %} public class Solution { public int minSubArrayLen(int s, int[] nums) { int slow=0,fast=0; int sum=0; int len=Integer.MAX_VALUE; while(fast=s) { if(len>fast-slow+1) len=fast-slow+1; break; } fast++; } while(slow<=fast) { sum-=nums[slow]; slow++; if(sum>=s) { if(len>fast-slow+1) len=fast-slow+1; } else break; } fast++; if(len==1) break; } return len==Integer.MAX_VALUE?0:len; } } {% endhighlight %}
