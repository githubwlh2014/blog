采用了迭代方法，而不是递归方法，注意后序遍历迭代时，“#”的标记作用

{% highlight ruby %} public class Solution { public List postorderTraversal(TreeNode root) { Deque stack=new LinkedList(); List list=new ArrayList(); if(root==null) return list; while(!stack.isEmpty()||root!=null) { if(root==null) { if(stack.peek() instanceof String) { stack.pop(); root=((TreeNode)stack.peek()).right; } else { list.add(((TreeNode)stack.pop()).val); } continue; } if(root.left!=null||root.right!=null) { stack.push(root); if(root.left!=null) { stack.push("#"); root=root.left; } else { root=root.right; } } else { list.add(root.val); root=null; } } return list; } } {% endhighlight %}
