递归求出左右子树的最低高度，然后选择左右子树最低高度较小的一个加一即为所求树的最低高度

{% highlight ruby %} public int minDepth(TreeNode root) { if(root==null) return 0; if(root.right==null&&root.left==null) return 1; int left=Integer.MAX_VALUE; //注意当左/右子树为空时，则左/右子树的最低高度为Integer.MAX_VALUE int right=Integer.MAX_VALUE; if(root.left!=null) left=minDepth(root.left)+1; //当左子树不为null时递归 if(root.right!=null) right=minDepth(root.right)+1; 当右子树不为null时递归 return left{{ page.date | date_to_string }}
