翻转字符串很简单

{% highlight ruby %} public class Solution { public String reverseVowels(String s) { if(s==null) return null; char[] str=s.toCharArray(); int start=0; int end=s.length()-1; Set set=new HashSet<>(Arrays.asList('a','o','e','i','u','E','A','O','I','U')); while(start=0&&!set.contains(str[end])) end--; if(start>=end) break; char tmp=str[start]; str[start]=str[end]; str[end]=tmp; start++; end--; } return new String(str); } } {% endhighlight %}
