利用先序遍历的思想，每次访问到一个节点时，把这个节点值存入List中，当访问到叶子节点时判断List中所有元素值和是否为sum ，如果等于Sum，保存此路径，否则返回父节点

{% highlight ruby %} public class PathSumII { private List> listSum=new ArrayList>(); public void pathSum(TreeNode root, int sum,List list) { if(root.left==null&&root.right==null) { if(sum-root.val==0) { list.add(root.val); listSum.add(new ArrayList(list)); list.remove(list.size()-1); //不要忘记返回时删除这一节点值 } return ; } list.add(root.val); sum-=root.val; if(root.left!=null) pathSum(root.left,sum,list); if(root.right!=null) pathSum(root.right,sum,list); list.remove(list.size()-1); //不要忘记返回时删除这一节点值 return; } public List> pathSum(TreeNode root, int sum) { listSum.clear(); if(root==null) return listSum; List list=new ArrayList(); pathSum(root,sum,list); return listSum; } } {% endhighlight %}
