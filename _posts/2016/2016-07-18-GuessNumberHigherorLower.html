二分查找

{% highlight ruby %} public class Solution extends GuessGame { public int guessNumber(int n) { if(n<1) return -1; int low=1; int high=n; int mid=low+(high-low)/2; while(low<=high) { if(guess(mid)==0) return mid; else if(guess(mid)==1) low=mid+1; else high=mid-1; mid=low+(high-low)/2; } return -1; } } {% endhighlight %}
