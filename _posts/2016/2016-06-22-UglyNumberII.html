求第n个ugly number，利用动态规划的思想，已知前k个ugly number，则第k+1个为2、3、5分别与前k个ugly number的乘积大于第k个ugly number的最小值

{% highlight ruby %} public class Solution { public int nthUglyNumber(int n) { if(n<=5) return n; int[] nums=new int[n+1]; int two=3,three=2,five=2; for(int i=1;i<6;i++) nums[i]=i; for(int i=6;i<=n;i++) { while(nums[two]*2<=nums[i-1]) two++; while(nums[three]*3<=nums[i-1]) three++; while(nums[five]*5<=nums[i-1]) five++; nums[i]=Math.min(nums[two]*2, Math.min(nums[three]*3, nums[five]*5)); } return nums[n]; } } {% endhighlight %}
