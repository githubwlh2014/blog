先排序然后再遍历数组，统计最大差值。（另外一种比较好的方法是桶排序）

{% highlight ruby %} public class Solution { public int maximumGap(int[] nums) { if(nums==null||nums.length==0) return 0; Arrays.sort(nums); int pre=nums[0]; int maxGap=0; for(int val:nums) { if(val-pre>maxGap) maxGap=val-pre; pre=val; } return maxGap; } } {% endhighlight %}
