动态规划，每次需要考虑中间间隔一个还是两个

{% highlight ruby %} public class Solution { public int rob(int[] nums) { if(nums==null||nums.length==0) return 0; if(nums.length==1) return nums[0]; int[] nums1=new int[nums.length+3]; System.arraycopy(nums, 0, nums1, 0, nums.length); for(int i=nums.length-1;i>=0;i--) { nums1[i]=nums1[i]+Math.max(nums1[i+2], nums1[i+3]); } int[] nums2=new int[nums.length+2]; System.arraycopy(nums,0,nums2,0,nums.length-1); for(int i=nums.length-2;i>=0;i--) { nums2[i]=nums2[i]+Math.max(nums2[i+2], nums2[i+3]); } return Math.max(nums2[0], Math.max(nums1[1], nums1[2])); } } {% endhighlight %}
