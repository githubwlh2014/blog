利用哈希Map和小顶堆

{% highlight ruby %} public class Solution { public List topKFrequent(int[] nums, int k) { Map map=new HashMap<>(); List list=new ArrayList<>(); for(int val:nums) { if(!map.containsKey(val)) map.put(val, 0); map.put(val, map.get(val)+1); } PriorityQueue queue=new PriorityQueue<>(); int i=0; for(int val:map.keySet()) { if(i=queue.peek()) list.add(val); } return list; } } {% endhighlight %}
