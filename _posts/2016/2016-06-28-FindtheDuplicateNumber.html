利用有环链表的思想进行求解

{% highlight ruby %} public class Solution { int findDuplicate(int[] nums) { if (nums.length > 1) { int slow = nums[0]; int fast = nums[nums[0]]; while (slow != fast) { slow = nums[slow]; fast = nums[nums[fast]]; } fast = 0; while (fast != slow) { fast = nums[fast]; slow = nums[slow]; } return slow; } return -1; } } {% endhighlight %}
