所有数组值与所有数组下标进行异或再与n进行异或，即可得到所求结果

{% highlight ruby %} public class Solution { public int missingNumber(int[] nums) { int xor = 0, i = 0; for (i = 0; i < nums.length; i++) { xor = xor ^ i ^ nums[i]; } return xor ^ i; } } {% endhighlight %}
