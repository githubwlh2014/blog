用leetcode141题的方法可以求出链表是否有环，并且可以计算出环中结点个数n，接着用两个引用，一个先走n步后，另外一个开始从头结点走两个引用同步走，相遇为所求结点

{% highlight ruby %} public class Solution { public ListNode detectCycle(ListNode head) { if(head==null) return null; ListNode back=head; ListNode front=head; do { if(front.next!=null&&front.next.next!=null) front=front.next.next; else return null; if(head.next!=null) head=head.next; else return null; } while(front!=head); int len=0; do { len++; if(front.next!=null&&front.next.next!=null) front=front.next.next; else return null; if(head.next!=null) head=head.next; else return null; } while(front!=head); front=back;head=back; while(len-->0) front=front.next; while(front!=back) { front=front.next; back=back.next; } return front; } } {% endhighlight %}
