利用位运算进行求解

{% highlight ruby %} public class Solution { public int getSum(int a, int b) { while(a!=0&&b!=0) { int tmp=((a&b)<<1); a=a^b; b=tmp; } return a==0?b:a; } } {% endhighlight %}
